# Project Structure

```
prototype/                        # Static site root (deployed to Cloudflare Pages)
├── index.html                    # Landing page (units section auto-generated by reconcile)
├── robots.txt                    # Welcomes AI crawlers
├── sitemap.xml                   # Auto-generated by reconcile
├── catalog.json                  # Auto-generated machine-readable index
├── .well-known/
│   ├── ai-knowledge.json         # AI discovery manifest (auto-generated by reconcile)
│   └── openapi-gpt.json          # OpenAI GPT Actions schema (OpenAPI 3.1.0)
├── _routes.json                  # Cloudflare Functions routing config
├── css/style.css
├── js/verified-date.js           # Client-side live date from Supabase (localStorage cached, 30-min TTL)
├── images/og-default.png         # Default OG share image (1200x630)
├── api/v1/openapi.json           # OpenAPI 3.0.3 spec (static, excluded from Functions)
├── consumer-electronics/         # Knowledge unit directories
│   ├── audio/
│   │   └── {topic}/{year}.html + .md
│   └── tv/
│       └── {topic}/{year}.html + .md
└── energy/us/...

functions/                        # Cloudflare Pages Functions (serverless)
├── _shared/
│   ├── utils.js                  # Shared utilities: CORS, jsonResponse, errorResponse, detectAgentType, logAccess, logError, getCatalog (60s TTL), searchCatalog, qualityStatus, computeEtag, checkNotModified, checkRateLimit, hashIp, normalizeQuestion, captureUnansweredQuery
│   └── validation.js             # Zod schemas: QuerySchema, BatchSchema, SuggestSchema, FeedbackSchema + validate() helper
├── go/[[slug]].js                # /go/{slug} → affiliate redirect (302) + error logging to pipeline_log
├── api/v1/
│   ├── query.js                  # GET /api/v1/query?q=...&limit=N&domain=...&region=...&jurisdiction=...&entity_type=... (auto-captures 0-result queries as suggestions)
│   ├── batch.js                  # POST /api/v1/batch — multi-query in single request (max 10), shared catalog parse
│   ├── suggest.js                # POST /api/v1/suggest — agents submit topic requests
│   ├── suggestions.js            # GET /api/v1/suggestions — list top unanswered suggestions
│   ├── feedback.js               # POST/GET /api/v1/feedback — report & browse content issues on knowledge units
│   ├── health.js                 # GET /api/v1/health — dependency status (catalog + supabase)
│   └── units/[[path]].js         # GET /api/v1/units/{id}.md or .json
└── mcp.js                        # POST /mcp — Streamable HTTP MCP endpoint (JSON-RPC 2.0, 6 tools, parity with npm v1.3.0)

mcp-server/                       # npm package: knowledgelib-mcp v1.3.0 (stdio MCP server, with tool annotations + batch_query)
├── cli.js                        # MCP server — tools: query_knowledge, batch_query, get_unit, list_domains, suggest_question, report_issue
├── package.json                  # Published to npm as knowledgelib-mcp
└── README.md

n8n-node/                         # npm package: n8n-nodes-knowledgelib v0.2.0 (n8n community node)
├── nodes/Knowledgelib/
│   └── Knowledgelib.node.ts      # Operations: Query, Get Unit, List Domains, Suggest Question, Report Issue
├── credentials/
│   └── KnowledgelibApi.credentials.ts
└── package.json                  # Published to npm as n8n-nodes-knowledgelib

langchain/                        # PyPI package: langchain-knowledgelib v0.2.0 (LangChain retriever)
├── langchain_knowledgelib/
│   ├── __init__.py               # Exports: KnowledgelibRetriever, suggest_question, asuggest_question, report_issue, areport_issue
│   └── retriever.py              # BaseRetriever + standalone suggest/report functions
├── tests/
└── pyproject.toml                # Published to PyPI as langchain-knowledgelib

knowledge_pipeline/
├── tracker.md                    # Pipeline status: pending/done per topic
├── reconcile.js                  # Regenerates catalog (v1.1), sitemap, index, ai-knowledge (v1.1), imports suggestions, feedback flagging, stats (direct DB)
├── reconcile_rest.js             # Same as reconcile.js but uses Supabase REST API (fallback)
├── indexnow.js                   # Search engine URL submission
├── migrate_seo_tags.js           # One-time migration: injects OG/Twitter/BreadcrumbList/viewport/creator into existing cards
├── migrate_improvement_plan.js   # Idempotent migration: temporal_validity, related_kos, constraints, jurisdiction, ai:* meta tags
├── extract_scripts.js            # Extracts long code blocks (>25 lines) from software .md into scripts/ subdirectory
└── templates/
    ├── product_comparison.md     # Markdown template for product cards (follow exactly)
    ├── product_comparison.html   # HTML template for product cards (follow exactly)
    ├── software_reference.md     # Markdown template for software/* cards (follow exactly)
    ├── software_reference.html   # HTML template for software/* cards (follow exactly)
    ├── fact.md                   # Markdown template for fact units (~200-400 tokens)
    ├── fact.html                 # HTML template for fact units
    ├── concept.md                # Markdown template for concept units (~400-800 tokens)
    ├── concept.html              # HTML template for concept units (DefinedTerm JSON-LD)
    ├── rule.md                   # Markdown template for rule units (~300-600 tokens)
    ├── rule.html                 # HTML template for rule units
    ├── erp_integration.md        # Markdown template for ERP integration cards
    ├── erp_integration.html      # HTML template for ERP integration cards
    ├── assessment.md             # Markdown template for assessment cards
    ├── assessment.html           # HTML template for assessment cards
    ├── benchmark.md              # Markdown template for benchmark cards
    ├── benchmark.html            # HTML template for benchmark cards
    ├── decision_framework.md     # Markdown template for decision framework cards
    ├── decision_framework.html   # HTML template for decision framework cards
    ├── playbook.md               # Markdown template for playbook cards
    ├── execution_recipe.md       # Markdown template for execution recipe cards
    ├── agent_prompt.md           # Markdown template for agent prompt cards
    ├── ai_knowledge_page.html    # HTML template for ai-knowledge discovery page
    └── sitemap.xml               # Sitemap template

db/
├── card_schema.js                # Shared Zod validation schema for card JSON (includes TemporalValiditySchema, RelatedKoSchema, InputNeededSchema, RuleScopeSchema)
├── insert_card.js                # Inserts card + affiliate links into Supabase (validates with Zod first)
├── insert_card_rest.js           # Same as insert_card.js but uses Supabase REST API (fallback)
├── run_migration.js              # Generic migration runner
├── 001_initial_schema.sql        # knowledge_cards + card_access_log + discovered_questions
├── 002_affiliate_tracking.sql    # affiliate_links + affiliate_click_log
├── 003_rls_policies.sql          # RLS for affiliate_links + affiliate_click_log
├── 004_set_pop_index.sql
├── 005_security_hardening.sql    # Status index + tightened click-log RLS policy
├── 006_pipeline_log.sql          # pipeline_log table for persistent observability
├── 008_question_suggestions.sql  # Extends discovered_questions: source tracking, RLS, upsert RPC, views
├── 009_fix_view_security.sql     # Fixes view security
├── 010_content_feedback.sql      # content_feedback table + RLS + indexes + summary view
└── pipeline_logger.js            # PipelineLogger class (start/complete/fail) used by insert + reconcile

wrangler.toml                     # Cloudflare Pages config
```
