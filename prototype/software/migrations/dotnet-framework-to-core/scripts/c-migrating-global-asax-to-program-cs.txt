// AFTER: Program.cs (.NET 8 minimal hosting)
var builder = WebApplication.CreateBuilder(args);

// Services (replaces FilterConfig, DI setup)
builder.Services.AddControllersWithViews(options =>
{
    options.Filters.Add<GlobalExceptionFilter>();
});
builder.Services.AddLogging();

var app = builder.Build();

// Middleware pipeline (replaces Global.asax events)
if (!app.Environment.IsDevelopment())
{
    app.UseExceptionHandler("/Home/Error");
    app.UseHsts();
}

app.UseHttpsRedirection();
app.UseStaticFiles();     // Replaces BundleConfig for static assets
app.UseRouting();
app.UseAuthentication();  // Replaces FormsAuthentication
app.UseAuthorization();

// Replaces RouteConfig.RegisterRoutes
app.MapControllerRoute(
    name: "default",
    pattern: "{controller=Home}/{action=Index}/{id?}");

app.Run();
